PROGRAM divmod;
{
  Written by: Joe Dorward
  Date: 08/11/00

  This program demonstrates the use of DIV and MOD
  to access the remainder in integer division
}

VAR
  first_number,
  second_number,
  the_answer,
  the_remainder : INTEGER;

BEGIN
  WRITE('Please enter an integer: ');
  READLN(first_number);

  WRITE('Please enter an integer: ');
  READLN(second_number);

  the_answer := first_number DIV second_number;
  the_remainder := first_number MOD second_number;

  WRITE('Dividing: ',first_number,' by: ',second_number);
  WRITE(' results in: ',the_answer);
  WRITELN(' remainder: ',the_remainder);
END.
